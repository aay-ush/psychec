107 template <class AstT>
108 struct ObserverInvoker
109 {
110  ObserverInvoker(VisitorObserver* o,
111  AstT* ast,
112  ScopeSymbol* scope) : o_(o), ast_(ast)
113  {
114  if (o_)
115  o_->enter(ast_, scope);
116  }
117 
118  ~ObserverInvoker()
119  {
120  if (o_)
121  o_->leave(ast_);
122  }
123 
124  VisitorObserver* o_;
125  AstT* ast_;
126 };
