300  struct Flags
301  {
302  // The token kind.
303  unsigned kind : 8;
304  // The token starts a new line.
305  unsigned newline : 1;
306  // The token is preceeded by whitespace(s).
307  unsigned whitespace : 1;
308  // The token is joined with the previous one.
309  unsigned joined : 1;
310  // The token originates from a macro expansion.
311  unsigned expanded : 1;
312  // The token originates from a macro expansion and does not correspond to an
313  // argument that went through substitution. Notice the example:
314  //
315  // #define FOO(a, b) a + b;
316  // FOO(1, 2)
317  //
318  // After preprocessing we would expect the following tokens: 1 + 2;
319  // Tokens '1', '+', '2', and ';' are all expanded. However only tokens '+' and ';'
320  // are generated.
321  unsigned generated : 1;
322  // The token is C++11 user-defined literal such as:
323  // 12_km, 0.5_Pa, 'c'_X, "abd"_L, u16"xyz"_M
324  unsigned userDefinedLiteral : 1;
325  // Unused...
326  unsigned pad : 2;
327  // The token length in bytes and UTF16 chars.
328  unsigned bytes : 16;
329  unsigned utf16chars : 16;
330  };
