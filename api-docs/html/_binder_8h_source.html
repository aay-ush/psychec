186 template <class SymT, class... SymTArgs>
187 std::unique_ptr<SymT> Binder::makeSymOrTySym(SymTArgs... args)
188 {
189  std::unique_ptr<SymT> sym(new SymT(tree_,
190  scopes_.top(),
191  syms_.top(),
192  std::forward<SymTArgs>(args)...));
193  return sym;
194 }

196 template <class SymT, class... SymTArgs>
197 void Binder::makeSymAndPushIt(const SyntaxNode* node, SymTArgs... args)
198 {
199  std::unique_ptr<SymT> sym = makeSymOrTySym<SymT>(std::forward<SymTArgs>(args)...);
200  pushSym(node, std::move(sym));
201 }

203 template <class SymT, class... SymTArgs>
204 void Binder::makeTySymAndPushIt(SymTArgs... args)
205 {
206  std::unique_ptr<SymT> sym = makeSymOrTySym<SymT>(std::forward<SymTArgs>(args)...);
207  pushTySym(std::move(sym));
